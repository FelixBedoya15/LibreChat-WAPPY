2. Firma Digital Táctil para Dispositivos Móviles y PC

Soporte completo con escucha nativo multitáctil para tabletas, celulares y clics de mouse. Calcula las fronteras del lienzo de forma dinámica: 

function initializeCanvas() {
    canvas = document.getElementById('signature-canvas');
    ctx = canvas.getContext('2d');
    ctx.strokeStyle = '#ffffff';
    ctx.lineWidth = 2.5;
    ctx.lineCap = 'round';
    ctx.lineJoin = 'round';
    
    canvas.addEventListener('mousedown', startDrawing);
    canvas.addEventListener('mousemove', draw);
    canvas.addEventListener('mouseup', stopDrawing);
    
    // Soporte Mobile/Táctil
    canvas.addEventListener('touchstart', startDrawingTouch, { passive: false });
    canvas.addEventListener('touchmove', drawTouch, { passive: false });
    canvas.addEventListener('touchend', stopDrawing);
}

function startDrawingTouch(e) {
    isDrawing = true;
    ctx.beginPath();
    const rect = canvas.getBoundingClientRect();
    const touch = e.touches[0];
    ctx.moveTo(touch.clientX - rect.left, touch.clientY - rect.top);
    e.preventDefault();
}

3. Registro Multimedia y Video Corto de 10 Segundos

Capacidad de adjuntar fotos (hasta 3), notas de audio de entrevistas, y un grabador de video rápido con cuenta regresiva estricta de 10 segundos para evidencias inmediatas de condiciones de seguridad: 

function startVideoRecording() {
    const btn = document.getElementById('btn-video-record');
    const player = document.getElementById('video-playback');
    const timer = document.getElementById('video-timer');
    
    btn.innerText = 'Grabando Video...';
    btn.disabled = true;
    timer.classList.remove('hidden');
    
    let seconds = 10;
    const interval = setInterval(() => {
        seconds--;
        timer.innerText = `00:0${seconds}`;
        if (seconds <= 0) {
            clearInterval(interval);
            btn.innerText = 'Grabar Video 10s';
            btn.disabled = false;
            timer.classList.add('hidden');
            player.classList.remove('hidden');
            alert("Video consolidado con éxito en el reporte.");
        }
    }, 1000);
}

4. Tablero Kanban de Planes de Acción y Seguimiento

Módulo ligero de gestión de tareas que permite priorizar, avanzar y documentar los planes correctivos y preventivos resultantes de la evaluación directamente en tres columnas: 

function renderKanban() {
    const columns = ['todo', 'progress', 'done'];
    columns.forEach(col => {
        const listEl = document.getElementById(`kanban-${col}`);
        listEl.innerHTML = '';
        const colTasks = currentWizard.tasks.filter(t => t.column === col);
        colTasks.forEach(task => {
            const el = document.createElement('div');
            el.className = 'p-3 bg-slate-900 border border-slate-800 rounded-xl flex items-center justify-between text-xs hover:bg-slate-850 cursor-pointer';
            el.innerHTML = `
                ${task.title}
                
            `;
            listEl.appendChild(el);
        });
    });
}

5. Chat Conversacional IA y Parseo de Corchetes para Inyección Directa

El copiloto interactúa mediante burbujas de diálogo completas. Si el usuario solicita a la IA redactar conclusiones, medidas o descripciones, la respuesta es provista en etiquetas `[Campo: Texto]` para que el usuario pueda aplicarlas con un clic: 

// Parseador e inyector automático
function applyAITextSuggestions(text) {
    const descRegex = /\[Descripción:\s*([\s\S]*?)\]/;
    const porquesRegex = /\[Porqués:\s*([\s\S]*?)\]/;
    const conclusionesRegex = /\[Conclusiones:\s*([\s\S]*?)\]/;

    const descMatch = text.match(descRegex);
    const porquesMatch = text.match(porquesRegex);
    const conclusionesMatch = text.match(conclusionesRegex);

    if (descMatch) document.getElementById('at-evento-desc').value = descMatch[1].trim();
    if (porquesMatch) document.getElementById('at-5-porques').value = porquesMatch[1].trim();
    if (conclusionesMatch) document.getElementById('at-conclusiones').value = conclusionesMatch[1].trim();
    
    saveLocalData();
}

6. Sistema de Exportación de Código Standalone

El aplicativo recupera su propia estructura HTML a través de peticiones locales e inyecta dinámicamente el estado actualizado en caliente antes de descargarse. Esto asegura que el archivo descargado contenga el logo, firmas corporativas y datos guardados de forma permanente: 

async function exportUpdatedHTML() {
    let baseHTML = "";
    try {
        const response = await fetch(window.location.href);
        if (response.ok) { baseHTML = await response.text(); }
    } catch (e) {
        baseHTML = "<!DOCTYPE html>\n<html>" + document.documentElement.innerHTML + "</html>";
    }

    const headerRegex = /let\s+appDocHeader\s*=\s*\{[\s\S]*?\}\s*;/;
    const newHeaderStr = `let appDocHeader = ${JSON.stringify(appDocHeader, null, 4)};`;

    let updatedHTML = baseHTML.replace(headerRegex, newHeaderStr);

    const blob = new Blob([updatedHTML], { type: 'text/html' });
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = url;
    a.download = appDocHeader.companyName.replace(/[^a-zA-Z0-9]/g, "_") + "_" + appDocHeader.changeCode + ".html";
    a.click();
}
