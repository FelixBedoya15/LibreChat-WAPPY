1. Estilos Adaptativos y Modo Oscuro

Todos los formatos deben soportar cambios de tema Claro/Oscuro dinámicos. Esto se realiza configurando Tailwind en modo class y aplicando transiciones suaves a fondos y bordes: 

<script>
    tailwind.config = {
        darkMode: 'class'
    }
</script>

<!-- Lógica de cambio de tema -->
function toggleTheme() {
    if (currentTheme === 'dark') {
        document.documentElement.classList.remove('dark');
        document.body.classList.replace('dark', 'light');
        currentTheme = 'light';
    } else {
        document.documentElement.classList.add('dark');
        document.body.classList.replace('light', 'dark');
        currentTheme = 'dark';
    }
    // Actualizar el color de los trazos de los canvas activos
    document.querySelectorAll('canvas').forEach(cvs => {
        const ctx = cvs.getContext('2d');
        ctx.strokeStyle = currentTheme === 'dark' ? '#ffffff' : '#1e293b';
    });
}

Las etiquetas principales del encabezado (Título del documento, nombre de la empresa, NIT, ARL, etc.) son contenteditable="true" para permitir la personalización de la identidad corporativa. Los cambios deben capturarse al perder el foco (onblur) y guardarse en el objeto appDocHeader:

    saveLocalData();
}

3. Persistencia de Medios Pesados (IndexedDB)

Para evadir la restricción estricta de 5MB de localStorage que genera fallos de tipo QuotaExceededError, se utiliza un almacenamiento híbrido: metadatos livianos en localStorage y archivos binarios pesados (imágenes, audios de evidencia, videos y logo corporativo de alta resolución) en una base de datos IndexedDB: 

// Definición del almacén IndexedDB
function openDB() {
    return new Promise((resolve, reject) => {
        const request = indexedDB.open('WappySSTDb', 2);
        request.onupgradeneeded = (e) => {
            const db = e.target.result;
            if (!db.objectStoreNames.contains('mediaStore')) {
                db.createObjectStore('mediaStore', { keyPath: 'id' });
            }
        };
        request.onsuccess = (e) => resolve(e.target.result);
        request.onerror = (e) => reject(e.target.error);
    });
}

async function saveMediaToDB(docId, photos, audio, video) {
    const db = await openDB();
    return new Promise((resolve, reject) => {
        const tx = db.transaction('mediaStore', 'readwrite');
        tx.objectStore('mediaStore').put({
            id: 'media_' + docId,
            photos: photos || [],
            audioBase64: audio || '',
            videoBase64: video || ''
        });
        tx.oncomplete = () => resolve();
        tx.onerror = () => reject(tx.error);
    });
}

4. Módulo de Firmas Multi-Pantalla

El módulo de firma digital incorpora tres funcionalidades cruciales para llevar a los formatos:

  • Soporte Táctil: Registro de los eventos táctiles touchstart, touchmove, y muy importante, touchend y touchcancel vinculados a la detención del trazo y guardado automático.
  • Carga de Archivo de Firma: Botón "Cargar Imagen" que lee archivos de imagen digitalizados, los escala y centra proporcionalmente al lienzo del canvas, y activa la persistencia.
  • Firmas Adicionales Dinámicas: Capacidad de añadir Validators extra al vuelo y conservarlos al exportar.
// Eventos de Dibujo Táctil
cvs.addEventListener('touchstart', (e) => {
    cvs.isDrawing = true;
    context.beginPath();
    const rect = cvs.getBoundingClientRect();
    const touch = e.touches[0];
    context.moveTo(touch.clientX - rect.left, touch.clientY - rect.top);
    e.preventDefault();
}, { passive: false });

cvs.addEventListener('touchmove', (e) => {
    if (!cvs.isDrawing) return;
    cvs.hasDrawn = true;
    const rect = cvs.getBoundingClientRect();
    const touch = e.touches[0];
    context.lineTo(touch.clientX - rect.left, touch.clientY - rect.top);
    context.stroke();
    e.preventDefault();
}, { passive: false });

cvs.addEventListener('touchend', stopDrawing);
cvs.addEventListener('touchcancel', stopDrawing);

// Carga de Imagen de Firma
function uploadSignatureImage(canvasId) {
    const fileInput = document.getElementById('upload-sig-' + canvasId);
    if (fileInput && fileInput.files[0]) {
        const reader = new FileReader();
        reader.onload = function(e) {
            const img = new Image();
            img.onload = function() {
                const cvs = document.getElementById(canvasId);
                const ctx = cvs.getContext('2d');
                ctx.clearRect(0, 0, cvs.width, cvs.height);
                const ratio = Math.min(cvs.width / img.width, cvs.height / img.height);
                const w = img.width * ratio;
                const h = img.height * ratio;
                ctx.drawImage(img, (cvs.width - w)/2, (cvs.height - h)/2, w, h);
                cvs.hasDrawn = true;
                saveLocalData();
            };
            img.src = e.target.result;
        };
        reader.readAsDataURL(fileInput.files[0]);
    }
}

5. Prevención de Condiciones de Carrera al Cargar Estado

Dado que cargar firmas desde Base64 mediante un objeto Image en JavaScript es una operación asíncrona (img.onload), debemos bloquear las operaciones de guardado automático (saveLocalData()) hasta que todos los lienzos hayan terminado de renderizarse, de lo contrario se sobrescribirán con canvas vacíos. 

// Lógica de hidratación asíncrona controlada
async function loadState() {
    isLoadingState = true; // Activa bandera global de bloqueo
    try {
        // ... cargar datos de texto y configurar tarjetas adicionales ...
        
        let imagesToLoad = 0;
        if (currentWizard.signatures) {
            Object.keys(currentWizard.signatures).forEach(canvasId => {
                const cvs = document.getElementById(canvasId);
                if (cvs && currentWizard.signatures[canvasId]) {
                    imagesToLoad++;
                    const img = new Image();
                    img.onload = function() {
                        cvs.getContext('2d').drawImage(img, 0, 0);
                        cvs.hasDrawn = true;
                        imagesToLoad--;
                        if (imagesToLoad === 0) isLoadingState = false; // Libera el bloqueo
                    };
                    img.onerror = () => {
                        imagesToLoad--;
                        if (imagesToLoad === 0) isLoadingState = false;
                    };
                    img.src = currentWizard.signatures[canvasId];
                }
            });
        }
        if (imagesToLoad === 0) {
            isLoadingState = false; // Libera inmediatamente si no hay firmas
        }
    } catch(e) {
        isLoadingState = false;
    }
}

6. Copiloto IA y Recopilación del Contexto del Formulario

Para que el chatbot interactivo en el panel lateral comprenda perfectamente el estado del documento, el sistema recopila dinámicamente el valor actual de todos los campos, testigos, averías y tareas en cada mensaje enviado al SDK de Gemini: 

function getDynamicContext() {
    let contextStr = `Empresa: ${appDocHeader.companyName}\nNIT: ${appDocHeader.companyNit}\n`;
    
    // Captura de entradas genéricas
    document.querySelectorAll('input[id], textarea[id], select[id]').forEach(input => {
        if (input.id && input.value) {
            contextStr += `${input.id.toUpperCase()}: ${input.value}\n`;
        }
    });
    
    // Captura de datos específicos (Testigos, Averías, Tareas Kanban)
    if (currentWizard.witnesses && currentWizard.witnesses.length > 0) {
        contextStr += "\n--- TESTIGOS REGISTRADOS ---\n";
        currentWizard.witnesses.forEach(w => {
            contextStr += `- ${w.name} (ID: ${w.doc}): "${w.testimony}"\n`;
        });
    }
    
    if (currentWizard.tasks && currentWizard.tasks.length > 0) {
        contextStr += "\n--- PLAN DE ACCIÓN (KANBAN) ---\n";
        currentWizard.tasks.forEach(t => {
            contextStr += `- Tarea: ${t.title} [Columna: ${t.column}]\n`;
        });
    }
    
    return contextStr;
}

8. Tablero Kanban de Planes de Acción

El componente de planes de acción implementa un tablero Kanban completamente dinámico y visual con columnas de estado (Por Hacer, En Progreso, Completado) y drag-and-drop nativo de HTML5. Las tareas se guardan en el array currentWizard.tasks. 

// Definición de Estructura de Tarea en Kanban
currentWizard.tasks = [
    {
        id: "task-1780084578181",
        title: "Capacitar personal en alturas",
        column: "todo" // 'todo', 'progress' o 'done'
    }
];

// Eventos Drag and Drop nativos
function drag(ev, taskId) {
    ev.dataTransfer.setData("text", taskId);
}
function allowDrop(ev) {
    ev.preventDefault();
}
function handleDrop(ev, columnName) {
    ev.preventDefault();
    const taskId = ev.dataTransfer.getData("text");
    const task = currentWizard.tasks.find(t => t.id === taskId);
    if (task) {
        task.column = columnName;
        renderKanban();
        saveLocalData();
    }
}

9. Matriz ATS Dinámica

La metodología ATS requiere desglosar las tareas operativas paso a paso, identificando los peligros y aplicando controles específicos. La tabla interactiva permite agregar, editar y remover filas mediante un array reactivo en currentWizard.atsRows. 

// Estructura de Fila ATS
currentWizard.atsRows = [
    {
        paso: "1",
        descripcion: "Montaje de andamios",
        peligros: "Caída de altura, colapso de estructura",
        controles: "Uso de arnés de seguridad anclado a línea de vida"
    }
];

// Renderizado dinámico de filas con campos vinculados
function renderATSTable() {
    const tbody = document.getElementById('ats-table-body');
    tbody.innerHTML = '';
    currentWizard.atsRows.forEach((row, index) => {
        tbody.innerHTML += `
            <tr>
                <td>${index + 1}</td>
                <td><textarea oninput="updateATSRow(${index}, 'descripcion', this.value)">${row.descripcion}</textarea></td>
                ...
            </tr>
        `;
    });
}
